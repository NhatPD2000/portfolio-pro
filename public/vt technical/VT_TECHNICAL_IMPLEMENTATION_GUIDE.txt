Bước 2: Cài đặt Odoo
Thời gian: 2 ngày

Commands:
  # Update system
  sudo apt update && sudo apt upgrade -y
  
  # Install dependencies
  sudo apt install python3-pip python3-dev libxml2-dev libxslt1-dev \
    zlib1g-dev libsasl2-dev libldap2-dev build-essential libssl-dev \
    libffi-dev libmysqlclient-dev libjpeg-dev libpq-dev libjpeg8-dev \
    liblcms2-dev libblas-dev libatlas-base-dev -y
  
  # Install PostgreSQL client
  sudo apt install postgresql-client -y
  
  # Install wkhtmltopdf (for PDF reports)
  sudo apt install wkhtmltopdf -y
  
  # Create Odoo user
  sudo adduser --system --home=/opt/odoo --group odoo
  
  # Install Odoo 16
  sudo su - odoo -s /bin/bash
  git clone https://www.github.com/odoo/odoo --depth 1 --branch 16.0 /opt/odoo/odoo16
  
  # Install Python dependencies
  pip3 install -r /opt/odoo/odoo16/requirements.txt
  
  # Create Odoo config file
  sudo nano /etc/odoo.conf
  
  # Start Odoo
  sudo systemctl start odoo
  sudo systemctl enable odoo

Kết quả:
  → Odoo 16 installed and running
  → Access: https://vttechnical-erp.com
